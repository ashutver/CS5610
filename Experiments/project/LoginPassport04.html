Adding functionality for adding training days

About the experiment

In this Experiment user can add training days to a workout. Once user is in workout details page here user can click on add training days and select day description and day to create a training day.

                  Server Side >>
                                       app.get("/api/days/:id", function (req, res) {
                                        var id = req.params.id;
                                        DayModel.find({ workoutid: id }, function (err, data) {
                                            res.json(data);
                                        });
                                    });
                                    app.post("/api/days/", function (req, res) {
                                        var day = req.body;
                                        var workoutid = day.workoutid;
                                        DayModel.create(day, function (err, data) {
                                            DayModel.find({ workoutid: workoutid }, function (err, data) {
                                                res.json(data);
                                            });
                                        });
                                    });
