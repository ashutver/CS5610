Code Snippet

                    Client side  >>
                      app.config(['$routeProvider',
                          function ($routeProvider) {
                              $routeProvider.
                                when('/developer', {
                                    templateUrl: 'developerApp/home.html',
                                    controller: 'homeController'
                                }).
                                when('/developer/:id', {
                                    templateUrl: 'developerApp/details.html',
                                    controller: 'detailsController'
                                }).
                                otherwise({
                                    redirectTo: '/developer'
                                });
                          }]);

                                           var userId = $rootScope.currentUser._id;
                                            $http.get("/api/workout/" + userId )
                                            .success(function (response) {
                                                $scope.workouts = response;
                                                console.log($scope.workouts);
                                                for (var w in $scope.workouts) {
                                                    var date = dateFormat(workouts[w].creationdate, "dd mmm yyyy");
                                                    workouts[w].creationdate = date;
                                                }
      
                                            });
                                            $scope.save = function () {
                                                var name = $("#inputText").val();
                                                console.log(name + " " + userId);
                                                var desc = { description: name };
                                                $http.post("/api/workout/" + userId, desc)
                                                .success(function (response) {
                                                    $scope.workouts = response;
                                                });
                                            }

                  Server Side >>
                                       app.get("/api/workout/:id", function (req, res) {
                                        var userid = req.params.id;
                                        WorkoutModel.find({ userid: userid }, function (err, data) {
                                            res.json(data);
                                           // console.log(data);
                                        });
                                    });
                                    app.post("/api/workout/:id", function (req, res) {
                                        var desc = req.body.description;
                                        console.log("workout " +desc);
                                        var userid1 = req.params.id;
                                        WorkoutModel.create({ description: desc, userid: userid1 }, function (err, data) {
                                            console.log("inserted");
                                            WorkoutModel.find({ userid: userid1 }, function (err, data) {
                                                res.json(data);
                                            });
                                        });
                                    });

Code Snippet Explanation

Here for a particular user we will get the workouts and display on screen. Each workout will be shown as a link and user can click on to proceed to detail page where user can add training days and exercises to it.
