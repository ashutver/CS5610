                        app.controller("ExerciseController",
                        function ($scope) {
                            var exercises = [
                                { title: "Bench Press", desc: "chest", category: "chest" },
                                { title: "Shoulder Press", desc: "shoulder", category: "shoulder" }
                            ];
                            $scope.exercises = exercises;
                            $scope.removeExercise = function (exercise) {
                                var index = $scope.exercises.indexOf(exercise);
                                $scope.exercises.splice(index, 1);
                            }

                <tr ng-repeat="exercise in exercises">
                <td>exercise.title</td>

Code Snippet Explanation

Here I am creating a array exercises and I am populating the data in to it. This is default data. Next, I will be assigning this to my scope variable. As there we will be placeholders on my UI to collect this data that is assigned to a scope variable so it will be rendered on UI. If I make any change to data in model then it will be reflected on UI.
